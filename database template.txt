CREATE TABLE Descriptions (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	description TEXT NOT NULL
);


CREATE TABLE Account (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	Name TEXT NOT NULL
);


CREATE TABLE Transactions (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	Date TEXT NOT NULL,
	Discription TEXT,
	Gas NUMERIC,
	Service NUMERIC,
	John NUMERIC,
	Medical NUMERIC,
	Misc NUMERIC,
	Tax NUMERIC,
	Savings NUMERIC,
);
