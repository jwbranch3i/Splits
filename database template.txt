CREATE TABLE Descriptions (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	description TEXT NOT NULL
);


CREATE TABLE Account (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	Name TEXT NOT NULL
);


CREATE TABLE Transactions (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	Date TEXT NOT NULL,
	Description_FK INTEGER NOT NULL,
	Account_FK INTEGER NOT NULL,
	CONSTRAINT Descriptions_FK FOREIGN KEY (id) REFERENCES Descriptions(id),
	CONSTRAINT Account_FK FOREIGN KEY (id) REFERENCES Account(id)
);
